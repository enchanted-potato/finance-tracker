import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest';
import MockAdapter from 'axios-mock-adapter';

// recharts mock BEFORE imports — SVG + ResizeObserver unavailable in happy-dom
vi.mock('recharts', () => ({
  ResponsiveContainer: ({ children }: { children: React.ReactNode }) => <div>{children}</div>,
  LineChart: ({ children }: { children: React.ReactNode }) => <div data-testid="line-chart">{children}</div>,
  Line: () => null,
  XAxis: () => null,
  YAxis: () => null,
  CartesianGrid: () => null,
  Tooltip: () => null,
  Legend: () => null,
  PieChart: ({ children }: { children: React.ReactNode }) => <div data-testid="pie-chart">{children}</div>,
  Pie: ({ children }: { children: React.ReactNode }) => <div>{children}</div>,
  Cell: () => null,
  BarChart: ({ children }: { children: React.ReactNode }) => <div data-testid="bar-chart">{children}</div>,
  Bar: () => null,
}));

import React from 'react';
import { screen, waitFor } from '@testing-library/react';
import { renderWithQuery } from './test-utils';
import { apiClient } from '@/lib/apiClient';
import { DashboardPage } from '@/pages/DashboardPage';

const mockAxios = new MockAdapter(apiClient);

const baseMockResponse = {
  cards: {
    net_worth: 150000,
    assets: 200000,
    liabilities: 50000,
    net_worth_delta: 1200,
  },
  trend: [
    { date: '2024-01-01', net_worth: 140000, assets: 190000, liabilities: 50000 },
    { date: '2024-02-01', net_worth: 150000, assets: 200000, liabilities: 50000 },
  ],
  allocation: [
    { name: 'ISA', value: 100000 },
    { name: 'Cash', value: 100000 },
  ],
  pension: [
    { name: 'Nest', value: 30000 },
  ],
};

describe('DashboardPage', () => {
  beforeEach(() => {
    mockAxios.reset();
  });
  afterEach(() => {
    vi.clearAllMocks();
  });

  it('renders metric card values from API (RDASH-01)', async () => {
    mockAxios.onGet('/api/dashboard').reply(200, baseMockResponse);
    renderWithQuery(<DashboardPage />);
    await waitFor(() => {
      expect(screen.getByText('£150,000')).toBeTruthy();
      expect(screen.getByText('£200,000')).toBeTruthy();
      expect(screen.getByText('£50,000')).toBeTruthy();
    });
  });

  it('shows loading state before data resolves (RDASH-01)', () => {
    // Do not call mockAxios.onGet — request stays pending
    renderWithQuery(<DashboardPage />);
    expect(screen.getByText(/loading dashboard/i)).toBeTruthy();
  });

  it('shows net_worth_delta in red when negative (RDASH-01)', async () => {
    const negative = {
      ...baseMockResponse,
      cards: { ...baseMockResponse.cards, net_worth_delta: -500 },
    };
    mockAxios.onGet('/api/dashboard').reply(200, negative);
    renderWithQuery(<DashboardPage />);
    await waitFor(() => {
      const delta = screen.getByText(/-£500/);
      expect(delta.className).toContain('text-red-400');
    });
  });

  it('shows net_worth_delta in green when positive (RDASH-01)', async () => {
    mockAxios.onGet('/api/dashboard').reply(200, baseMockResponse);
    renderWithQuery(<DashboardPage />);
    await waitFor(() => {
      const delta = screen.getByText(/\+£1,200/);
      expect(delta.className).toContain('text-green-400');
    });
  });

  it('renders em-dash for null card values (RDASH-01)', async () => {
    const nullCards = {
      ...baseMockResponse,
      cards: { net_worth: null, assets: null, liabilities: null, net_worth_delta: null },
    };
    mockAxios.onGet('/api/dashboard').reply(200, nullCards);
    renderWithQuery(<DashboardPage />);
    await waitFor(() => {
      const dashes = screen.getAllByText('—');
      expect(dashes.length).toBeGreaterThanOrEqual(3);
    });
  });

  it('renders trend line chart container (RDASH-02)', async () => {
    mockAxios.onGet('/api/dashboard').reply(200, baseMockResponse);
    renderWithQuery(<DashboardPage />);
    await waitFor(() => {
      expect(screen.getByTestId('line-chart')).toBeTruthy();
    });
  });

  it('renders asset allocation pie chart container (RDASH-03)', async () => {

    renderWithQuery(<DashboardPage />);

    });
  });
});
