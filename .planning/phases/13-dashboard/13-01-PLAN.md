```typescript
import { useQuery } from '@tanstack/react-query';
import {
  ResponsiveContainer,
  LineChart,
  Line,
  XAxis,
  YAxis,
  CartesianGrid,
  Tooltip,
  Legend,
  PieChart,
  Pie,
  Cell,
  BarChart,
  Bar,
} from 'recharts';
import { dashboardApi } from '@/lib/api/dashboard';
import type { DashboardResponse } from '@/lib/api/dashboard';

const GBP = new Intl.NumberFormat('en-GB', {
  style: 'currency',
  currency: 'GBP',
  maximumFractionDigits: 0,
});

const GBP_AXIS = new Intl.NumberFormat('en-GB');

function formatMoney(v: number | null): string {
  if (v === null || v === undefined) return '‚Äî';
  return GBP.format(v);
}

const ALLOCATION_COLORS = [
  '#3b82f6',
  '#10b981',
  '#f59e0b',
  '#8b5cf6',
  '#ef4444',
  '#06b6d4',
];

export function DashboardPage() {
  const { data, isPending, isError } = useQuery<DashboardResponse>({
    queryKey: ['dashboard'],
    queryFn: dashboardApi.get,
  });

  if (isPending) {
    return (
      <div className="p-6">
        <p className="text-muted-foreground">Loading dashboard‚Ä¶</p>
      </div>
    );
  }

  if (isError || !data) {
    return (
      <div className="p-6">
        <p className="text-destructive">Failed to load dashboard data.</p>
      </div>
    );
  }

  const { cards, trend, allocation, pension } = data;

  const deltaColor =
    cards.net_worth_delta !== null && cards.net_worth_delta < 0
      ? 'text-red-400'
      : 'text-green-400';

  return (
    <div className="p-6 space-y-6">
      <h1 className="text-2xl font-semibold">Dashboard</h1>

      {/* Metric Cards */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
        {/* Net Worth */}
        <div className="rounded-xl p-5 bg-blue-950/50 border border-blue-800/50">
          <p className="text-sm text-muted-foreground mb-1">Net Worth</p>
          <p className="text-2xl font-bold">{formatMoney(cards.net_worth)}</p>
          {cards.net_worth_delta !== null && (
            <p className={`text-sm mt-1 ${deltaColor}`}>
              {cards.net_worth_delta >= 0 ? '+' : ''}
              {formatMoney(cards.net_worth_delta)} this month
            </p>
          )}
        </div>

        {/* Assets */}
        <div className="rounded-xl p-5 bg-green-950/50 border border-green-800/50">
          <p className="text-sm text-muted-foreground mb-1">Assets</p>
          <p className="text-2xl font-bold">{formatMoney(cards.assets)}</p>
        </div>

        {/* Liabilities */}
        <div className="rounded-xl p-5 bg-red-950/50 border border-red-800/50">
          <p className="text-sm text-muted-foreground mb-1">Liabilities</p>
          <p className="text-2xl font-bold">{formatMoney(cards.liabilities)}</p>
        </div>
      </div>

      {/* Trend Line Chart */}
      <div className="rounded-xl border border-border bg-card p-5">
        <h2 className="text-base font-semibold mb-4">Net Worth Trend</h2>
        <ResponsiveContainer width="100%" height={280}>
          <LineChart data={trend}>
            <CartesianGrid strokeDasharray="3 3" stroke="#334155" />
            <XAxis dataKey="date" tick={{ fontSize: 12 }} />
            <YAxis
              tickFormatter={(v) => GBP_AXIS.format(v)}
              tick={{ fontSize: 12 }}
              width={80}
            />
            <Tooltip
              formatter={(value: number) => GBP.format(value)}
            />
            <Legend />
            <Line
              type="monotone"
              dataKey="net_worth"
              stroke="#3b82f6"
              dot={false}
              name="Net Worth"
            />
            <Line
              type="monotone"
              dataKey="assets"
              stroke="#10b981"
              dot={false}
              name="Assets"
            />
            <Line
              type="monotone"
              dataKey="liabilities"
              stroke="#ef4444"
              dot={false}
              name="Liabilities"
            />
          </LineChart>
        </ResponsiveContainer>
      </div>

      {/* Bottom row: Allocation + Pension side by side */}
      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        {/* Asset Allocation Donut */}
        <div className="rounded-xl border border-border bg-card p-5">
          <h2 className="text-base font-semibold mb-4">Asset Allocation</h2>
          <ResponsiveContainer width="100%" height={260}>
            <PieChart>
              <Pie
                data={allocation}
                dataKey="value"
                nameKey="name"
                innerRadius={70}
                outerRadius={110}
              >
                {allocation.map((_entry, index) => (
                  <Cell
                    key={`cell-${index}`}
                    fill={ALLOCATION_COLORS[index % ALLOCATION_COLORS.length]}
                  />
                ))}
              </Pie>
              <Tooltip formatter={(value: number) => GBP.format(value)} />
              <Legend />
            </PieChart>
          </ResponsiveContainer>
        </div>

        {/* Pension Bar Chart */}
        <div className="rounded-xl border border-border bg-card p-5">
          <h2 className="text-base font-semibold mb-4">Pension by Provider</h2>
          <ResponsiveContainer width="100%" height={260}>
            <BarChart data={pension}>
              <CartesianGrid strokeDasharray="3 3" stroke="#334155" />
              <XAxis dataKey="name" tick={{ fontSize: 12 }} />
              <YAxis
                tickFormatter={(v) => GBP_AXIS.format(v)}
                tick={{ fontSize: 12 }}
                width={80}
              />
              <Tooltip formatter={(value: number) => GBP.format(value)} />
              <Bar dataKey="value" fill="#6366f1" name="Balance" />
            </BarChart>
          </ResponsiveContainer>
        </div>
      </div>
    </div>
  );
}
```
